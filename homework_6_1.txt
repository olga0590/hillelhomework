let n = 12;//задаю количество рандомных числел
let arr = [];
function randomNumbers(start, end) {
  for (let i = 1; i <= n; i++) {
    let random = Math.round(Math.random() * (end - start) + start);
    arr.push(random);
  }
  return arr;
}
console.log(randomNumbers(10, 100));

//Определение сколько парный и непарный:
let count = 0;
let oddCount = 0;
for (let i = 0; i < arr.length; i++) {
  if (arr[i] % 2 === 0) {
    count++;
  } else {
    oddCount++;
  }
}
console.log(`Парных числе в массиве: ${count}`);
console.log(`Непарных числе в массиве: ${oddCount}`);

//Процентное соотношение парных и непарных числе в массиве
const percent = Math.round((count / arr.length) * 100);
console.log(`Парных чисел в массиве: ${percent}%`);
const oddPercent = 100 - percent;
console.log(`Непарных чисел в массиве: ${oddPercent}%`);
